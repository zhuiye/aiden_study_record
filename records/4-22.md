解法 Two 循环

```js
function missingNumber(arr) {
  let min = 0;

  for (let i = 0; i < arr.length + 1; i++) {
    let flag = false;
    for (let value of arr) {
      if (min === value) {
        flag = true;
        break;
      }
    }
    if (!flag) {
      return min;
    }
    min++;
  }
}
```

构建 hash 循环

牺牲一定的空间,换取时间

```js
function missingNumber(arr) {
  let hash = {};
  for (let i = 0; i < arr.length; i++) {
    hash[arr[i]] = i;
  }

  for (let i = 0; i < arr.length + 1; i++) {
    if (hash[i] === undefined) {
      return i;
    }
  }
}
```
